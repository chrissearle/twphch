365         Challenge challenge = (Challenge) query.getSingleResult();
366 
367         // Sum up votes
368         List<Image> images = challenge.getImages();
369 
370         for (Image image : images) {
371             image.setFinalVoteCount((long) image.getVotes().size());
372             em.persist(image);
373 
374             if (image.getId().equals("VoteTestImage1")) {
375                 assert image.getFinalVoteCount() == 2 : "Incorrect number of votes for image 1" + image.getFinalVoteCount();
376             }
377 
378             if (image.getId().equals("VoteTestImage2")) {
379                 assert image.getFinalVoteCount() == 0 : "Incorrect number of votes for image 2" + image.getFinalVoteCount();
380             }
381 
382             if (image.getId().equals("VoteTestImage3")) {
