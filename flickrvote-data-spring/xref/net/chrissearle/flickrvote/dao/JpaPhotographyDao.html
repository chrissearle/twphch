19  import net.chrissearle.common.jpa.JpaDao;
20  import net.chrissearle.flickrvote.model.Photographer;
21  import org.apache.log4j.Logger;
22  import org.springframework.stereotype.Repository;
23  
24  import javax.persistence.NoResultException;
25  import javax.persistence.Query;
26  import java.util.List;
27  
28  /**
29   * Class JpaPhotographyDao implements PhotographyDao using JPA
30   *
31   * @author chris
32   */
33  @Repository
34  public class JpaPhotographyDao extends JpaDao<String, Photographer> implements PhotographyDao {
35      private Logger log = Logger.getLogger(JpaPhotographyDao.class);
36  
37      /**
38       * Method findByUsername finds the photographer with the given username. Null if none found.
39       *
40       * @param username of type String
41       * @return Photographer
42       */
43      public Photographer findByUsername(String username) {
44          if (log.isDebugEnabled()) {
45              log.debug("findByUsername : " + username);
46          }
47  
48          Query query = entityManager.createQuery("select p from Photographer p where p.username = :username");
49          query.setParameter("username", username);
50  
51          try {
52              return (Photographer) query.getSingleResult();
53          } catch (NoResultException e) {
54              if (log.isDebugEnabled()) {
55                  log.debug("No matching user found");
56              }
57  
58              // Just means that there is no photographer yet present
59              return null;
60          }
61      }
62  
63      /**
64       * Method findByToken finds the photographer with the given token. Null if none found.
65       *
66       * @param token of type String
67       * @return Photographer
68       */
69      public Photographer findByToken(String token) {
70          Query query = entityManager.createQuery("select p from Photographer p where p.token = :token");
71          query.setParameter("token", token);
72  
73          try {
74              return (Photographer) query.getSingleResult();
75          } catch (NoResultException e) {
76              // Just means that there is no photographer yet validated with flickr
77              return null;
78          }
79      }
80  
81      /**
82       * Method persist saves the photographer if new, updates fullname and token if existing.
83       *
84       * @param photographer of type Photographer
85       */
86      @Override
87      public void persist(Photographer photographer) {
88          Photographer p = findByUsername(photographer.getUsername());
89  
90          if (p != null) {
91              p.setFullname(photographer.getFullname());
92              p.setToken(photographer.getToken());
93  
94              if (photographer.getId() != null) {
95                  p.setAdministrator(photographer.isAdministrator());
96              }
97  
98              photographer = p;
99          }
100 
101         super.persist(photographer);
102     }
103 
104     /**
105      * Method clearVotes removes all votes from the entire system.
106      */
107     public void clearVotes() {
108         Query query = entityManager.createQuery("DELETE FROM Vote v");
109         query.executeUpdate();
110     }
111 
112     /**
113      * Method all returns all photographers
114      *
115      * @return List<Photographer>
116      */
117     @SuppressWarnings("unchecked")
118     public List<Photographer> all() {
119         Query query = entityManager.createQuery("SELECT p FROM Photographer p");
120 
121         return (List<Photographer>) query.getResultList();
122     }
