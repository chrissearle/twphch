17  package net.chrissearle.flickrvote.flickr.impl;
18  
19  import com.aetrion.flickr.Flickr;
20  import com.aetrion.flickr.FlickrException;
21  import com.aetrion.flickr.RequestContext;
22  import com.aetrion.flickr.auth.Auth;
23  import com.aetrion.flickr.auth.AuthInterface;
24  import com.aetrion.flickr.photos.comments.CommentsInterface;
25  import net.chrissearle.flickrvote.flickr.CommentDAO;
26  import net.chrissearle.flickrvote.flickr.FlickrServiceException;
27  import org.constretto.annotation.Configuration;
28  import org.constretto.annotation.Configure;
29  import org.springframework.beans.factory.annotation.Autowired;
30  import org.springframework.stereotype.Component;
31  import org.xml.sax.SAXException;
32  
33  import java.io.IOException;
34  import java.util.logging.Level;
35  import java.util.logging.Logger;
36  
37  @Component
38  public class FlickrJCommentDAO implements CommentDAO {
39      private Logger logger = Logger.getLogger(this.getClass().getName());
40      private boolean commentsActiveFlag;
41      private String adminAuthToken;
42  
43      private Flickr flickr;
44  
45      @Autowired
46      public FlickrJCommentDAO(Flickr flickr) {
47          this.flickr = flickr;
48      }
49  
50      @Configure
51      public void configure(@Configuration(expression = "flickr.admin.auth.token") String adminToken,
52                            @Configuration(expression = "flickr.admin.auth.active") Boolean adminActive) {
53          this.commentsActiveFlag = adminActive;
54          this.adminAuthToken = adminToken;
55      }
56  
57      public void postComment(String imageId, String comment) {
58          if (logger.isLoggable(Level.FINE)) {
59              logger.fine("Posting comment check: " + commentsActiveFlag);
60          }
61  
62          if (commentsActiveFlag) {
63              if (logger.isLoggable(Level.INFO)) {
64                  logger.info("Image commented: " + imageId);
65  
66                  if (logger.isLoggable(Level.FINE)) {
67                      logger.fine("Commenting on : " + imageId + " with comment: " + comment);
68                  }
69              }
70  
71              comment(imageId, comment);
72          }
73  
74      }
75  
76      private void comment(String imageId, String comment) {
77          try {
78              setAuthInContext();
79  
80              postToFlickr(imageId, comment);
81          } catch (IOException e) {
82              throw new FlickrServiceException(e);
83          } catch (SAXException e) {
84              throw new FlickrServiceException(e);
85          } catch (FlickrException e) {
86              throw new FlickrServiceException(e);
87          }
88      }
89  
90      private void postToFlickr(String imageId, String comment) throws IOException, SAXException, FlickrException {
91          CommentsInterface commentsInterface = flickr.getCommentsInterface();
92  
93          commentsInterface.addComment(imageId, comment);
94      }
95  
96      private void setAuthInContext() throws IOException, SAXException, FlickrException {
97          RequestContext context = RequestContext.getRequestContext();
98  
99          context.setAuth(getAdminAuth());
100     }
101 
102     private Auth getAdminAuth() throws IOException, SAXException, FlickrException {
103         AuthInterface authInterface = flickr.getAuthInterface();
104 
105         return authInterface.checkToken(adminAuthToken);
106     }
107 }
