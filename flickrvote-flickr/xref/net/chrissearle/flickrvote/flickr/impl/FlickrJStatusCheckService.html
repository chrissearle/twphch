20  import net.chrissearle.flickrvote.flickr.model.FlickrImage;

23  import net.chrissearle.flickrvote.flickr.model.FlickrPhotographer;
24  import org.springframework.beans.factory.annotation.Autowired;
25  import org.springframework.stereotype.Service;

27  import java.util.*;
28  
29  @Service
30  public class FlickrJStatusCheckService implements FlickrStatusCheckService {
31      ImageTagSearchDAO imageTagSearchDAO;
32      ImageDAO imageDAO;
33      UserDAO userDAO;
34  
35      @Autowired
36      public FlickrJStatusCheckService(ImageTagSearchDAO imageTagSearchDAO, UserDAO userDAO, ImageDAO imageDAO) {
37          this.imageTagSearchDAO = imageTagSearchDAO;
38          this.userDAO = userDAO;
39          this.imageDAO = imageDAO;
40      }
41  
42      public Set<FlickrImageStatus> checkSearch(String tag, Date earliestDate) {
43          if (earliestDate == null) {
44              throw new FlickrServiceException("Earliest date cannot be null");
45          }
46  
47          Set<FlickrImageStatus> issues = new HashSet<FlickrImageStatus>();
48  
49          Map<String, FlickrImageStatus> seenPhotographers = new HashMap<String, FlickrImageStatus>();
50  
51          FlickrImages images = imageTagSearchDAO.searchTag(tag);
52  
53          for (FlickrImage image : images.getImages()) {
54              final String photographerId = image.getPhotographer().getFlickrId();
55  
56              image.setPhotographer(userDAO.getUser(photographerId));
57  
58              if (checkImageForDateIssue(earliestDate, image, issues)) {
59                  continue;
60              }
61  
62              createImageMultipleIssue(seenPhotographers, image, photographerId);
63          }
64  
65          for (FlickrImageStatus status : seenPhotographers.values()) {
66              if (status.getImageCount() > 1) {
67                  issues.add(status);
68              }
69          }
70  
71  
72          return issues;
73      }
74  
75      @Override
76      public FlickrImageStatus checkSearch(String tag, Date earliestDate, String imageId) {
77          Set<FlickrImageStatus> issues = new HashSet<FlickrImageStatus>();
78  
79          if (checkImageForDateIssue(earliestDate, imageDAO.getImage(imageId), issues)) {
80              return issues.iterator().next();
81          } else {
82              return null;
83          }
84      }
85  
86  
87      private void createImageMultipleIssue(Map<String, FlickrImageStatus> seenPhotographers, FlickrImage image, String photographerId) {
88          if (haveSeenPhotographer(seenPhotographers, photographerId)) {
89              FlickrImageStatus status = new FlickrImageStatus(FlickrImageStatus.ImageStatus.MULTIPLE_IMAGES,
90                      image.getPhotographer(), image);
91  
92              seenPhotographers.put(photographerId, status);
93          } else {
94              seenPhotographers.get(photographerId).addImage(image);
95          }
96      }
97  
98      private boolean haveSeenPhotographer(Map<String, FlickrImageStatus> seenPhotographers, String photographerId) {
99          return !seenPhotographers.containsKey(photographerId);
100     }
101 
102     private boolean checkImageForDateIssue(Date earliestDate, FlickrImage image, Set<FlickrImageStatus> issues) {
103         if (checkImageOnOrAfterDate(earliestDate, image)) {
104             FlickrImageStatus status = new FlickrImageStatus(FlickrImageStatus.ImageStatus.TAKEN_DATE,
105                     image.getPhotographer(), image);
106 
107             issues.add(status);
108 
109             return true;
110         }
111 
112         return false;
113     }
114 
115     private boolean checkImageOnOrAfterDate(Date earliestDate, FlickrImage image) {
116         return image.getTakenDate() != null && image.getTakenDate().getTime() < earliestDate.getTime();
117     }
