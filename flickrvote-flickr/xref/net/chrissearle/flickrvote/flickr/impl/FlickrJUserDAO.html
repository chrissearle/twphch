17  package net.chrissearle.flickrvote.flickr.impl;
18  
19  import com.aetrion.flickr.Flickr;
20  import com.aetrion.flickr.FlickrException;
21  import com.aetrion.flickr.people.PeopleInterface;
22  import com.aetrion.flickr.people.User;
23  import net.chrissearle.flickrvote.flickr.FlickrServiceException;
24  import net.chrissearle.flickrvote.flickr.UserDAO;
25  import net.chrissearle.flickrvote.flickr.model.FlickrPhotographer;
26  import org.springframework.beans.factory.annotation.Autowired;
27  import org.springframework.stereotype.Component;
28  import org.xml.sax.SAXException;
29  
30  import java.io.IOException;
31  import java.util.logging.Level;
32  import java.util.logging.Logger;
33  
34  @Component
35  public class FlickrJUserDAO implements UserDAO {
36      private Logger logger = Logger.getLogger(this.getClass().getName());
37      private Flickr flickr;
38  
39      @Autowired
40      public FlickrJUserDAO(Flickr flickr) {
41          this.flickr = flickr;
42      }
43  
44      public FlickrPhotographer getUser(String id) {
45          try {
46              return retrieveAndBuildPhotographer(id);
47          } catch (SAXException e) {
48              throw new FlickrServiceException(e);
49          } catch (FlickrException e) {
50              throw new FlickrServiceException(e);
51          } catch (IOException e) {
52              throw new FlickrServiceException(e);
53          }
54      }
55  
56      private FlickrPhotographer retrieveAndBuildPhotographer(String id) throws IOException, SAXException, FlickrException {
57          User user = retrieveUser(id);
58  
59          if (logger.isLoggable(Level.INFO)) {
60              logger.info("User info fetched for " + user.getUsername());
61          }
62  
63          return buildPhotographer(user);
64      }
65  
66      private FlickrPhotographer buildPhotographer(User user) {
67          return new FlickrPhotographer(user.getId(), user.getUsername(), user.getRealName(), user.getBuddyIconUrl());
68      }
69  
70      private User retrieveUser(String id) throws IOException, SAXException, FlickrException {
71          PeopleInterface peopleInterface = flickr.getPeopleInterface();
72  
73          User user = peopleInterface.getInfo(id);
74  
75          if (user == null) {
76              throw new FlickrServiceException("No user found with ID: " + id);
77          }
78  
79          return user;
80      }
81  }
