47      public void prepare() throws Exception {
48          ChallengeSummary challengeSummary = null;
49  
50          // Get a list of all open challenges
51          Set<ChallengeSummary> challenges = challengeService.getChallengesByType(ChallengeType.OPEN);
52  
53          // Currently the web layer is designed for one current challengeSummary only
54          if (challenges.size() > 0) {
55              challengeSummary = challenges.iterator().next();
56          }
57  
58          if (log.isDebugEnabled()) {
59              log.debug("Current challengeSummary " + challengeSummary);
60          }
61  
62          if (challengeSummary != null) {
63              challenge = new DisplayChallengeSummary(challengeSummary);
64  
65              ChallengeItem challengeItem = photographyService.getChallengeImages(challengeSummary.getTag());
66  
67              images = new ArrayList<DisplayImage>(challengeItem.getImages().size());
68  
69              for (ImageItem image : challengeItem.getImages()) {
70                  images.add(new DisplayImage(image));
71              }
72  
73              Collections.sort(images, new Comparator<DisplayImage>() {
74                  public int compare(DisplayImage o1, DisplayImage o2) {
75                      return o2.getPostedDate().compareTo(o1.getPostedDate());
76                  }
77              });
78          }
79      }
