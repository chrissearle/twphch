17  package net.chrissearle.flickrvote.web.admin;
18  
19  import com.opensymphony.xwork2.ActionSupport;
20  import net.chrissearle.flickrvote.service.PhotographyService;
21  import net.chrissearle.flickrvote.web.FlickrVoteWebConstants;
22  import net.chrissearle.flickrvote.web.model.DisplayPhotographer;
23  import net.chrissearle.flickrvote.web.model.Photographer;
24  import org.apache.struts2.interceptor.SessionAware;
25  import org.springframework.beans.factory.annotation.Autowired;
26  
27  import java.util.Map;
28  import java.util.logging.Level;
29  import java.util.logging.Logger;
30  
31  public class LoginAsPhotographer extends ActionSupport implements SessionAware {
32      private Logger logger = Logger.getLogger(LoginAsPhotographer.class.getName());
33  
34      private Map<String, Object> session;
35  
36      private String id;
37  
38      @Autowired
39      private PhotographyService service;
40  
41      @Override
42      public void setSession(Map<String, Object> stringObjectMap) {
43          this.session = stringObjectMap;
44      }
45  
46      @Override
47      public String execute() throws Exception {
48          Photographer admin = (Photographer)session.get(FlickrVoteWebConstants.FLICKR_USER_SESSION_KEY);
49  
50          Photographer photographer = new DisplayPhotographer(service.findById(id));
51  
52          session.put(FlickrVoteWebConstants.FLICKR_USER_SESSION_KEY, photographer);
53  
54          if (logger.isLoggable(Level.INFO)) {
55              logger.info("Admin " + admin.getPhotographerName() + " just logged in as " + photographer.getPhotographerName());
56          }
57  
58          return SUCCESS;
59      }
60  
61      public String getId() {
62          return id;
63      }
64  
65      public void setId(String id) {
66          this.id = id;
67      }
68  }
