17  package net.chrissearle.flickrvote.service.cli;
18  
19  import twitter4j.Twitter;
20  import twitter4j.TwitterException;
21  import twitter4j.TwitterFactory;
22  import twitter4j.http.AccessToken;
23  import twitter4j.http.RequestToken;
24  
25  import java.io.BufferedReader;
26  import java.io.IOException;
27  import java.io.InputStreamReader;
28  
29  public class TwitterOAuthCLI {
30      public void authorize(String key, String secret) throws TwitterException, IOException {
31          Twitter twitter = new TwitterFactory().getInstance();
32  
33          twitter.setOAuthConsumer(key, secret);
34  
35          RequestToken requestToken = twitter.getOAuthRequestToken();
36  
37          AccessToken accessToken = null;
38  
39          BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
40  
41          while (null == accessToken) {
42              System.out.println("Open the following URL and grant access to your account:");
43              System.out.println(requestToken.getAuthorizationURL());
44              System.out.print("Enter the PIN(if available) or just hit enter.[PIN]:");
45  
46              String pin = br.readLine();
47              try {
48                  if (pin.length() > 0) {
49                      accessToken = twitter.getOAuthAccessToken(requestToken, pin);
50                  } else {
51                      accessToken = twitter.getOAuthAccessToken();
52                  }
53  
54                  System.out.println("Token: " + accessToken.getToken());
55                  System.out.println("TokenSecret: " + accessToken.getTokenSecret());
56              } catch (TwitterException te) {
57                  if (401 == te.getStatusCode()) {
58                      System.out.println("Unable to get the access token.");
59                  } else {
60                      te.printStackTrace();
61                  }
62              }
63          }
64      }
65  
66      public static void main(String[] args) throws TwitterException, IOException {
67          TwitterOAuthCLI app = new TwitterOAuthCLI();
68  
69          app.authorize(args[0], args[1]);
70      }
