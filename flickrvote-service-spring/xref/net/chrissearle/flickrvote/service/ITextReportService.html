63      public void generateHistoryReport() {
64          if (logger.isInfoEnabled()) {
65              logger.info("Generating history report");
66          }
67  
68          try {
69              Document document = new Document();
70  
71              File tempFile = File.createTempFile("flickrvote_history", ".pdf");
72  
73              PdfWriter pdfWriter = PdfWriter.getInstance(document, new FileOutputStream(tempFile));
74  
75              setFileMetadata(pdfWriter);
76  
77              document.open();
78  
79              String reportTitle = challengeMessageService.getHistoryReportTitle();
80  
81              document.addTitle(reportTitle);
82  
83              document.add(new Paragraph(reportTitle, new Font(Font.TIMES_ROMAN, 36, Font.BOLD)));
84  
85              addChallenges(document);
86  
87              document.close();
88  
89              if (logger.isInfoEnabled()) {
90                  logger.info("Renaming " + tempFile.getAbsolutePath() + " to " + historyReportPath);
91              }
92  
93              File pdfFile = new File(historyReportPath);
94              if (pdfFile.exists()) {
95                  if (!pdfFile.delete()) {
96                      if (logger.isEnabledFor(Level.WARN)) {
97                          logger.warn("Could not delete file " + historyReportPath);
98                      }
99                  }
100             }
101 
102             if (pdfFile.canWrite()) {
103                 if (!tempFile.renameTo(pdfFile)) {
104                     if (logger.isEnabledFor(Level.WARN)) {
105                         logger.warn("Could not rename file " + tempFile.getAbsolutePath() + " to " + historyReportPath);
106                     }
107                 }
108             } else {
109                 if (logger.isEnabledFor(Level.WARN)) {
110                     logger.warn("Cannot write to file " + historyReportPath);
111                 }
112             }
113         } catch (DocumentException e) {
114             if (logger.isEnabledFor(Level.WARN)) {
115                 logger.warn("Unable to create PDF", e);
116             }
117         } catch (IOException e) {
118             if (logger.isEnabledFor(Level.WARN)) {
119                 logger.warn("Unable to create PDF", e);
120             }
121         }
122     }

124     private void addChallenges(Document document) throws DocumentException {
125         List<ChallengeSummary> challenges = new ArrayList<ChallengeSummary>(challengeService.getChallengesByType(ChallengeType.CLOSED));
126 
127         Collections.sort(challenges, new Comparators.ChallengeSummarySortByStartDateComparator());
128 
129         int chapterCount = 1;
130         for (ChallengeSummary challenge : challenges) {
131             addChallengeChapter(document, chapterCount++, challenge);
132         }
133     }

135     private void addChallengeChapter(Document document, int chapterCount, ChallengeSummary challenge) throws DocumentException {
136         Chapter challengeChapter = new Chapter(new Paragraph("#" + challenge.getTag(), new Font(Font.TIMES_ROMAN, 24, Font.BOLD)), chapterCount);
137         challengeChapter.setNumberDepth(0);
138         challengeChapter.setBookmarkOpen(true);
139         challengeChapter.setBookmarkTitle("#" + challenge.getTag() + " " + challenge.getTitle());
140 
141         challengeChapter.add(new Paragraph(challenge.getTitle(), new Font(Font.TIMES_ROMAN, 20, Font.BOLDITALIC)));
142 
143         addInfoSection(challengeChapter, challenge);
144 
145         addImagesSection(challengeChapter, challenge);
146 
147         document.add(challengeChapter);
148     }

150     private void addImagesSection(Chapter challengeChapter, ChallengeSummary challenge) {
151         challengeChapter.newPage();
152 
153         ChallengeItem challengeDetails = photographyService.getChallengeImages(challenge.getTag());
154 
155         List<ImageItem> images = new ArrayList<ImageItem>(challengeDetails.getImages());
156 
157         Collections.sort(images, new Comparators.ImageItemSortByVoteCount());
158 
159         String title = challengeMessageService.getImageSectionTitle();
160 
161         Section section = challengeChapter.addSection(new Paragraph(title, SECTION_FONT), 0);
162         section.setBookmarkTitle(title);
163         section.setBookmarkOpen(false);
164 
165         for (ImageItem image : images) {
166             addImage(section, image);
167         }
168     }

170     private void addImage(Section section, ImageItem image) {
171 
172         String title = image.getRank() + ": " + image.getTitle();
173         Section subSection = section.addSection(new Paragraph(title, SUBSECTION_FONT), 0);
174         subSection.setBookmarkOpen(true);
175         subSection.setBookmarkTitle(title);
176 
177         try {
178             Image img = Image.getInstance(new URL(image.getImageUrl()));
179             img.setAlignment(Image.ALIGN_MIDDLE);
180             subSection.add(img);
181 
182             float[] widths = {1f, 3f};
183             PdfPTable table = new PdfPTable(widths);
184             table.setHorizontalAlignment(Element.ALIGN_CENTER);
185             table.getDefaultCell().setVerticalAlignment(Element.ALIGN_MIDDLE);
186             table.getDefaultCell().setPadding(5);
187             table.setSpacingBefore(15);
188 
189             DateFormat df = new SimpleDateFormat("yyyy-MM-dd");
190 
191             addTableRow(table, challengeMessageService.getHistoryImageTitleTitle(), image.getTitle());
192             addTableRow(table, challengeMessageService.getHistoryImageRankTitle(), "" + image.getRank());
193             addTableRow(table, challengeMessageService.getHistoryImageVoteTitle(), "" + image.getVoteCount());
194             if (image.getPostedDate() != null) {
195                 addTableRow(table, challengeMessageService.getHistoryImagePostedTitle(), df.format(image.getPostedDate()));
196             }
197             addTableRow(table, challengeMessageService.getHistoryImageUrlTitle(), image.getUrl());
198             addTableRow(table, challengeMessageService.getHistoryImagePhotographerTitle(), image.getPhotographer().getName());
199 
200             subSection.add(table);
201 
202             subSection.newPage();
203         } catch (BadElementException e) {
204             if (logger.isEnabledFor(Level.WARN)) {
205                 logger.warn("Unable to add image: " + image, e);
206             }
207         } catch (IOException e) {
208             if (logger.isEnabledFor(Level.WARN)) {
209                 logger.warn("Unable to add image: " + image, e);
210             }
211         }
212     }

219     private void addInfoSection(Chapter challengeChapter, ChallengeSummary challenge) {
220         String title = challengeMessageService.getInfoSectionTitle();
221 
222         Section infoSection = challengeChapter.addSection(new Paragraph(title, SECTION_FONT), 0);
223         infoSection.setBookmarkOpen(false);
224         infoSection.setBookmarkTitle(title);

226         PdfPTable table = new PdfPTable(1);
227         table.setHorizontalAlignment(Element.ALIGN_CENTER);
228         table.getDefaultCell().setBorder(PdfPCell.NO_BORDER);
229         table.getDefaultCell().setVerticalAlignment(Element.ALIGN_MIDDLE);
230         table.getDefaultCell().setPadding(5);
231         table.setSpacingBefore(15);
232         table.setSpacingAfter(15);
233 
234         table.addCell(new Paragraph(challengeMessageService.getInfoStartDate(challenge.getStartDate()), new Font(Font.TIMES_ROMAN, 14, Font.NORMAL)));
235         table.addCell(new Paragraph(challengeMessageService.getInfoVoteDate(challenge.getVoteDate()), new Font(Font.TIMES_ROMAN, 14, Font.NORMAL)));
236         table.addCell(new Paragraph(challengeMessageService.getInfoEndDate(challenge.getEndDate()), new Font(Font.TIMES_ROMAN, 14, Font.NORMAL)));
237 
238         infoSection.add(table);
239 
240         try {
241             Image img = Image.getInstance(new URL("http://vote.twphch.com/twitterphotochallenge/chart/showChart.action?tag=" + challenge.getTag()));
242             img.setAlignment(Image.ALIGN_MIDDLE);
243             img.scalePercent(50);
244             infoSection.add(img);
245         } catch (BadElementException e) {
246             if (logger.isEnabledFor(Level.WARN)) {
247                 logger.warn("Unable to add chart: " + challenge.getTag(), e);
248             }
249         } catch (IOException e) {
250             if (logger.isEnabledFor(Level.WARN)) {
251                 logger.warn("Unable to add chart: " + challenge.getTag(), e);
252             }
253         }

264     public byte[] getHistoryReport() {
265         long length = getHistoryReportSize();
266 
267         InputStream in = null;
268 
269         try {
270             // Apparently arrays are max bound by int not long.
271             if (length > 0 && length < Integer.MAX_VALUE) {
272 
273                 in = new FileInputStream(new File(historyReportPath));
274 
275                 byte[] bytes = new byte[(int) length];
276 
277                 int offset = 0;
278                 int numRead = 0;
279                 while (offset < bytes.length
280                         && (numRead = in.read(bytes, offset, bytes.length - offset)) >= 0) {
281                     offset += numRead;
282                 }
283 
284                 if (offset < bytes.length) {
285                     if (logger.isEnabledFor(Level.ERROR)) {
286                         logger.error("Failed to read history file");
287                     }
288                 } else {
289                     in.close();
290                     return bytes;
291                 }
292             }
293         } catch (FileNotFoundException e) {
294             if (logger.isEnabledFor(Level.ERROR)) {
295                 logger.error("Failed to find history file", e);
296             }
297         } catch (IOException e) {
298             if (logger.isEnabledFor(Level.ERROR)) {
299                 logger.error("File error reading history file", e);
300             }
301         } finally {
302             if (in != null) {
303                 try {
304                     in.close();
305                 } catch (IOException e) {
306                     if (logger.isEnabledFor(Level.WARN)) {
307                         logger.warn("Unable to close stream in finally block");
308                     }
309                 }
310             }
311         }
312         return new byte[0];
313     }

315     public long getHistoryReportSize() {
316         File file = new File(historyReportPath);
317 
318         if (!file.exists() || !file.canRead()) {
319             return REPORT_UNAVAILABLE;
320         }
321 
322         return file.length();
323     }
