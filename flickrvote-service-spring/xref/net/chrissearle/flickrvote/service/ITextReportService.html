48      public void generateHistoryReport() {
49          if (logger.isInfoEnabled()) {
50              logger.info("Generating history report");
51          }
52  
53          try {
54              Document document = new Document();
55  
56              File tempFile = File.createTempFile("flickrvote_history", ".pdf");
57  
58              PdfWriter pdfWriter = PdfWriter.getInstance(document, new FileOutputStream(tempFile));
59  
60              setFileMetadata(pdfWriter);
61  
62              document.open();
63  
64              String reportTitle = challengeMessageService.getHistoryReportTitle();
65  
66              document.addTitle(reportTitle);
67  
68              document.add(new Paragraph(reportTitle, new Font(Font.TIMES_ROMAN, 36, Font.BOLD)));
69  
70              addChallenges(document);
71  
72              document.close();
73  
74              if (logger.isInfoEnabled()) {
75                  logger.info("Renaming " + tempFile.getAbsolutePath() + " to " + historyReportPath);
76              }
77  
78              File pdfFile = new File(historyReportPath);
79              if (pdfFile.exists()) {
80                  if (!pdfFile.delete()) {
81                      if (logger.isEnabledFor(Level.WARN)) {
82                          logger.warn("Could not delete file " + historyReportPath);
83                      }
84                  }
85              }
86  
87              if (pdfFile.canWrite()) {
88                  if (!tempFile.renameTo(pdfFile)) {
89                      if (logger.isEnabledFor(Level.WARN)) {
90                          logger.warn("Could not rename file " + tempFile.getAbsolutePath() + " to " + historyReportPath);
91                      }
92                  }
93              } else {
94                  if (logger.isEnabledFor(Level.WARN)) {
95                      logger.warn("Cannot write to file " + historyReportPath);
96                  }
97              }
98          } catch (DocumentException e) {
99              if (logger.isEnabledFor(Level.WARN)) {
100                 logger.warn("Unable to create PDF", e);
101             }
102         } catch (IOException e) {
103             if (logger.isEnabledFor(Level.WARN)) {
104                 logger.warn("Unable to create PDF", e);
105             }
106         }

109     private void addChallenges(Document document) throws DocumentException {
110         List<ChallengeSummary> challenges = new ArrayList<ChallengeSummary>(challengeService.getChallengesByType(ChallengeType.CLOSED));
111 
112         Collections.sort(challenges, new Comparator<ChallengeSummary>() {
113             public int compare(ChallengeSummary o1, ChallengeSummary o2) {
114                 return o1.getStartDate().compareTo(o2.getStartDate());
115             }
116         });
117 
118         int chapterCount = 1;
119         for (ChallengeSummary challenge : challenges) {
120             chapterCount = addChallengeChapter(document, chapterCount++, challenge);
121         }
122     }

131 
132         addInfoSection(challengeChapter, challenge);

134         addImagesSection(challengeChapter, challenge);

164     private void addImage(Section section, ImageItem image) {
165 
166         String title = image.getRank() + ": " + image.getTitle();
167         Section subSection = section.addSection(new Paragraph(title, SUBSECTION_FONT), 0);
168         subSection.setBookmarkOpen(true);

175 
176             float[] widths = {1f, 3f};
177             PdfPTable table = new PdfPTable(widths);
178             table.setHorizontalAlignment(Element.ALIGN_CENTER);
179             table.getDefaultCell().setVerticalAlignment(Element.ALIGN_MIDDLE);
180             table.getDefaultCell().setPadding(5);
181             table.setSpacingBefore(15);
182 
183             DateFormat df = new SimpleDateFormat("yyyy-MM-dd");
184 
185             addTableRow(table, challengeMessageService.getHistoryImageTitleTitle(), image.getTitle());
186             addTableRow(table, challengeMessageService.getHistoryImageRankTitle(), "" + image.getRank());
187             addTableRow(table, challengeMessageService.getHistoryImageVoteTitle(), "" + image.getVoteCount());
188             if (image.getPostedDate() != null) {
189                 addTableRow(table, challengeMessageService.getHistoryImagePostedTitle(), df.format(image.getPostedDate()));
190             }
191             addTableRow(table, challengeMessageService.getHistoryImageUrlTitle(), image.getUrl());
192             addTableRow(table, challengeMessageService.getHistoryImagePhotographerTitle(), image.getPhotographer().getName());
193 
194             subSection.add(table);
195 
196             subSection.newPage();
197         } catch (BadElementException e) {
198             if (logger.isEnabledFor(Level.WARN)) {
199                 logger.warn("Unable to add image: " + image, e);
200             }
201         } catch (IOException e) {
202             if (logger.isEnabledFor(Level.WARN)) {
203                 logger.warn("Unable to add image: " + image, e);

258     public byte[] getHistoryReport() {
259         long length = getHistoryReportSize();
260 
261         InputStream in = null;
262 
263         try {
264             // Apparently arrays are max bound by int not long.
265             if (length > 0 && length < Integer.MAX_VALUE) {
266 
267                 in = new FileInputStream(new File(historyReportPath));
268 
269                 byte[] bytes = new byte[(int) length];
270 
271                 int offset = 0;
272                 int numRead = 0;
273                 while (offset < bytes.length
274                         && (numRead = in.read(bytes, offset, bytes.length - offset)) >= 0) {
275                     offset += numRead;
276                 }
277 
278                 if (offset < bytes.length) {
279                     if (logger.isEnabledFor(Level.ERROR)) {
280                         logger.error("Failed to read history file");
281                     }
282                 } else {
283                     in.close();
284                     return bytes;
285                 }
286             }
287         } catch (FileNotFoundException e) {
288             if (logger.isEnabledFor(Level.ERROR)) {
289                 logger.error("Failed to find history file", e);
290             }
291         } catch (IOException e) {
292             if (logger.isEnabledFor(Level.ERROR)) {
293                 logger.error("File error reading history file", e);
294             }
295         } finally {
296             if (in != null) {
297                 try {
298                     in.close();
299                 } catch (IOException e) {
300                     if (logger.isEnabledFor(Level.WARN)) {
301                         logger.warn("Unable to close stream in finally block");
302                     }
303                 }
304             }
305         }
306         return new byte[0];
307     }
