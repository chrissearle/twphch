47      public void generateHistoryReport() {
48          if (logger.isInfoEnabled()) {
49              logger.info("Generating history report");
50          }
51  
52          try {
53              Document document = new Document();
54  
55              File tempFile = File.createTempFile("flickrvote_history", ".pdf");
56  
57              PdfWriter pdfWriter = PdfWriter.getInstance(document, new FileOutputStream(tempFile));
58  
59              setFileMetadata(pdfWriter);
60  
61              document.open();
62  
63              String reportTitle = challengeMessageService.getHistoryReportTitle();
64  
65              document.addTitle(reportTitle);
66  
67              document.add(new Paragraph(reportTitle, new Font(Font.TIMES_ROMAN, 36, Font.BOLD)));
68  
69              addChallenges(document);
70  
71              document.close();
72  
73              if (logger.isInfoEnabled()) {
74                  logger.info("Renaming " + tempFile.getAbsolutePath() + " to " + historyReportPath);
75              }
76  
77              File pdfFile = new File(historyReportPath);
78              if (pdfFile.exists()) {
79                  if (!pdfFile.delete()) {
80                      if (logger.isEnabledFor(Level.WARN)) {
81                          logger.warn("Could not delete file " + historyReportPath);
82                      }
83                  }
84              }
85  
86              if (pdfFile.canWrite()) {
87                  if (!tempFile.renameTo(pdfFile)) {
88                      if (logger.isEnabledFor(Level.WARN)) {
89                          logger.warn("Could not rename file " + tempFile.getAbsolutePath() + " to " + historyReportPath);
90                      }
91                  }
92              } else {
93                  if (logger.isEnabledFor(Level.WARN)) {
94                      logger.warn("Cannot write to file " + historyReportPath);
95                  }
96              }
97          } catch (DocumentException e) {
98              if (logger.isEnabledFor(Level.WARN)) {
99                  logger.warn("Unable to create PDF", e);
100             }
101         } catch (IOException e) {
102             if (logger.isEnabledFor(Level.WARN)) {
103                 logger.warn("Unable to create PDF", e);
104             }
105         }
106     }

108     private void addChallenges(Document document) throws DocumentException {
109         List<ChallengeSummary> challenges = new ArrayList<ChallengeSummary>(challengeService.getChallengesByType(ChallengeType.CLOSED));
110 
111         Collections.sort(challenges, new Comparators.ChallengeSummarySortByStartDateComparator());

154     private void addImage(Section section, ImageItem image) {
155 
156         String title = image.getRank() + ": " + image.getTitle();
157         Section subSection = section.addSection(new Paragraph(title, SUBSECTION_FONT), 0);
158         subSection.setBookmarkOpen(true);
159         subSection.setBookmarkTitle(title);
160 
161         try {
162             Image img = Image.getInstance(new URL(image.getImageUrl()));
163             img.setAlignment(Image.ALIGN_MIDDLE);
164             subSection.add(img);
165 
166             float[] widths = {1f, 3f};
167             PdfPTable table = new PdfPTable(widths);
168             table.setHorizontalAlignment(Element.ALIGN_CENTER);
169             table.getDefaultCell().setVerticalAlignment(Element.ALIGN_MIDDLE);
170             table.getDefaultCell().setPadding(5);
171             table.setSpacingBefore(15);
172 
173             DateFormat df = new SimpleDateFormat("yyyy-MM-dd");
174 
175             addTableRow(table, challengeMessageService.getHistoryImageTitleTitle(), image.getTitle());
176             addTableRow(table, challengeMessageService.getHistoryImageRankTitle(), "" + image.getRank());
177             addTableRow(table, challengeMessageService.getHistoryImageVoteTitle(), "" + image.getVoteCount());
178             if (image.getPostedDate() != null) {
179                 addTableRow(table, challengeMessageService.getHistoryImagePostedTitle(), df.format(image.getPostedDate()));
180             }
181             addTableRow(table, challengeMessageService.getHistoryImageUrlTitle(), image.getUrl());
182             addTableRow(table, challengeMessageService.getHistoryImagePhotographerTitle(), image.getPhotographer().getName());
183 
184             subSection.add(table);
185 
186             subSection.newPage();
187         } catch (BadElementException e) {
188             if (logger.isEnabledFor(Level.WARN)) {
189                 logger.warn("Unable to add image: " + image, e);
190             }
191         } catch (IOException e) {
192             if (logger.isEnabledFor(Level.WARN)) {
193                 logger.warn("Unable to add image: " + image, e);
194             }
195         }

248     public byte[] getHistoryReport() {
249         long length = getHistoryReportSize();
250 
251         InputStream in = null;
252 
253         try {
254             // Apparently arrays are max bound by int not long.
255             if (length > 0 && length < Integer.MAX_VALUE) {
256 
257                 in = new FileInputStream(new File(historyReportPath));
258 
259                 byte[] bytes = new byte[(int) length];
260 
261                 int offset = 0;
262                 int numRead = 0;
263                 while (offset < bytes.length
264                         && (numRead = in.read(bytes, offset, bytes.length - offset)) >= 0) {
265                     offset += numRead;
266                 }
267 
268                 if (offset < bytes.length) {
269                     if (logger.isEnabledFor(Level.ERROR)) {
270                         logger.error("Failed to read history file");
271                     }
272                 } else {
273                     in.close();
274                     return bytes;
275                 }
276             }
277         } catch (FileNotFoundException e) {
278             if (logger.isEnabledFor(Level.ERROR)) {
279                 logger.error("Failed to find history file", e);
280             }
281         } catch (IOException e) {
282             if (logger.isEnabledFor(Level.ERROR)) {
283                 logger.error("File error reading history file", e);
284             }
285         } finally {
286             if (in != null) {
287                 try {
288                     in.close();
289                 } catch (IOException e) {
290                     if (logger.isEnabledFor(Level.WARN)) {
291                         logger.warn("Unable to close stream in finally block");
292                     }
293                 }
294             }
295         }
296         return new byte[0];
297     }
