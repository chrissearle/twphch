113     /**
114      * Method retrieveAndStorePhotographer causes flickr to grab a photographer and will save/update the local db.
115      *
116      * @param id - flickr ID
117      * @return the photographer retrieved. Null if none found.
118      */
119     public PhotographerItem retrieveAndStorePhotographer(String id) {
120         // Check to see if present
121         Photographer photographer = photographyDao.findById(id);
122 
123         FlickrPhotographer flickrPhotographer = flickrService.getUserByFlickrId(id);
124 
125         if (photographer == null) {
126             photographer = new Photographer(flickrPhotographer.getToken(), flickrPhotographer.getUsername(),
127                     flickrPhotographer.getRealname(), flickrPhotographer.getFlickrId(), flickrPhotographer.getIconUrl());
128         } else {
129             photographer.setUsername(flickrPhotographer.getUsername());
130             photographer.setFullname(flickrPhotographer.getRealname());
131             photographer.setIconUrl(flickrPhotographer.getIconUrl());
132         }
133 
134         photographyDao.persist(photographer);
135 
136         return new PhotographerItemInstance(photographer);
137     }

170     /**
171      * Method getChallengeImages retrieves all images for a challenge,
172      *
173      * @param tag of type String
174      * @return ChallengeItem
175      */
176     public ChallengeItem getChallengeImages(String tag) {
177         Set<ImageItem> images = new HashSet<ImageItem>();
178 
179         Challenge challenge = challengeDao.findByTag(tag);
180 
181         if (challenge == null) {
182             return null;
183         }
184 
185         Map<String, ImageItem> seenImages = new HashMap<String, ImageItem>();
186 
187         if (challenge.getVotingState() == ChallengeState.OPEN) {
188             // Grab images from flickr
189             for (FlickrImage image : flickrService.searchImagesByTag(tag, challenge.getStartDate())) {
190                 ImageItemInstance imageItem = new ImageItemInstance(image);
191                 images.add(imageItem);
192                 seenImages.put(image.getPhotographer().getFlickrId(), imageItem);
193             }
194         }
195 
196         // Grab images from db. Merge in if flickr images have already been fetched
197         for (Image image : challenge.getImages()) {
198             final String photographerId = image.getPhotographer().getId();
199 
200             if (seenImages.containsKey(photographerId)) {
201                 // The db has an image for a photographer that has been found by search.
202                 // This only happens when administrators add an image manually.
203                 // In this case - use the db image.
204                 images.remove(seenImages.get(photographerId));
205             }
206 
207             images.add(new ImageItemInstance(image));
208         }
209 
210         return new ChallengeItemInstance(challenge.getTag(), challenge.getName(), images);
211     }

233     /**
234      * Method retrieveAndStoreImage causes a flickr retrieval of the image (and photographer). Saves/updates to the local db.
235      *
236      * @param id              flickr ID of image
237      * @param tag             challenge tag
238      * @param replaceExisting - if true will remove an image if retrieving an image with the same photographer for the same challenge.
239      * @return ImageItem - null if no image found
240      */
241     protected ImageItem retrieveAndStoreImage(String id, String tag, boolean replaceExisting) {
242         Challenge challenge = challengeDao.findByTag(tag);
243 
244         if (challenge == null)
245             return null;
246 
247         Image image = imageDao.findById(id);
248 
249         FlickrImage flickrImage = flickrService.getImageByFlickrId(id);
250 
251         if (image == null) {
252             if (logger.isInfoEnabled()) {
253                 logger.info("Storing image " + flickrImage.getFlickrId());
254             }
255             String photographerId = flickrImage.getPhotographer().getFlickrId();
256             Photographer photographer = photographyDao.findById(photographerId);
257 
258             Image existing = checkForExistingImage(photographer, tag);
259 
260             if (existing != null) {
261                 // Found an existing image for this challenge by this photographer
262                 if (replaceExisting) {
263                     // Normal functionality - remove the existing one - we're doing an update
264                     photographer.removeImage(existing);
265                     imageDao.remove(existing);
266                 } else {
267                     // Special case - e.g. we don't want to replace admin overridden images at freeze.
268                     return new ImageItemInstance(existing);
269                 }
270             }
271 
272             image = new Image();
273             image.setId(flickrImage.getFlickrId());
274             image.setMediumImage(flickrImage.getImageUrl());
275             image.setPage(flickrImage.getUrl());
276             image.setTitle(flickrImage.getTitle());
277             image.setPostedDate(flickrImage.getPostedDate());
278 
279 
280             if (photographer == null) {
281                 retrieveAndStorePhotographer(photographerId);
282 
283                 photographer = photographyDao.findById(photographerId);
284             }
285 
286             photographer.addImage(image);
287 
288             photographyDao.persist(photographer);
289 
290             challenge.addImage(image);
291 
292             challengeDao.persist(challenge);
293         } else {
294             if (logger.isInfoEnabled()) {
295                 logger.info("Refreshing image " + flickrImage.getFlickrId());
296             }
297 
298             image.setTitle(flickrImage.getTitle());
299             image.setPostedDate(flickrImage.getPostedDate());
300             image.setPage(flickrImage.getUrl());
301             image.setMediumImage(flickrImage.getImageUrl());
302 
303             imageDao.persist(image);
304         }
305 
306         return new ImageItemInstance(image);
307     }

309     /**
310      * Method checkForExistingImage checks to see if the photographer has an image for the given challenge
311      *
312      * @param photographer of type Photographer
313      * @param tag          of type String
314      * @return Image
315      */
316     private Image checkForExistingImage(Photographer photographer, String tag) {
317         for (Image image : photographer.getImages()) {
318             if (image.getChallenge().getTag().equals(tag)) {
319                 return image;
320             }
321         }
322         return null;
323     }

363     /**
364      * Method freezeChallenge - takes the current challenge - performs a flickr search and stores the results locally.
365      */
366     public void freezeChallenge() {
367         if (logger.isInfoEnabled()) {
368             logger.info("Freezing challenge");
369         }
370 
371         Challenge challenge = challengeDao.getVotingChallenge();
372 
373         if (challenge == null) {
374             return;
375         }
376 
377         if (logger.isInfoEnabled()) {
378             logger.info("Freezing challenge : " + challenge);
379         }
380 
381 
382         List<FlickrImage> images = flickrService.searchImagesByTag(challenge.getTag(), challenge.getStartDate());
383 
384         for (FlickrImage image : images) {
385             retrieveAndStoreImage(image.getFlickrId(), challenge.getTag(), false);
386         }
387     }
