118     /**
119      * Method retrieveAndStorePhotographer causes flickr to grab a photographer and will save/update the local db.
120      *
121      * @param id - flickr ID
122      * @return the photographer retrieved. Null if none found.
123      */
124     public PhotographerItem retrieveAndStorePhotographer(String id) {
125         // Check to see if present
126         Photographer photographer = photographyDao.findById(id);
127 
128         FlickrPhotographer flickrPhotographer = flickrUserDao.getUser(id);
129 
130         if (photographer == null) {
131             photographer = new Photographer(flickrPhotographer.getUsername(),
132                     flickrPhotographer.getRealname(), flickrPhotographer.getFlickrId(), flickrPhotographer.getIconUrl());
133         } else {
134             photographer.setUsername(flickrPhotographer.getUsername());
135             photographer.setFullname(flickrPhotographer.getRealname());
136             photographer.setIconUrl(flickrPhotographer.getIconUrl());
137         }
138 
139         photographyDao.persist(photographer);
140 
141         return new PhotographerItemInstance(photographer);
142     }

227     /**
228      * Method retrieveAndStoreImage causes a flickr retrieval of the image (and photographer). Saves/updates to the local db.
229      *
230      * @param id              flickr ID of image
231      * @param tag             challenge tag
232      * @param replaceExisting - if true will remove an image if retrieving an image with the same photographer for the same challenge.
233      * @return ImageItem - null if no image found
234      */
235     protected ImageItem retrieveAndStoreImage(String id, String tag, boolean replaceExisting, String photographerCheckId) {
236         Challenge challenge = challengeDao.findByTag(tag);
237 
238         if (challenge == null)
239             return null;
240 
241         Image image = imageDao.findById(id);
242 
243         FlickrImage flickrImage = flickrImageDao.getImage(id);
244 
245         if (image == null) {
246             if (logger.isLoggable(Level.INFO)) {
247                 logger.info("Storing image " + flickrImage.getFlickrId());
248             }
249             String photographerId = flickrImage.getPhotographer().getFlickrId();
250 
251             if (photographerCheckId == null || photographerCheckId.equals(photographerId)) {
252 
253                 Photographer photographer = photographyDao.findById(photographerId);
254 
255                 if (photographer == null) {
256                     retrieveAndStorePhotographer(photographerId);
257 
258                     photographer = photographyDao.findById(photographerId);
259                 }
260 
261                 Image existing = checkForExistingImage(photographer, tag);
262 
263                 if (existing != null) {
264                     // Found an existing image for this challenge by this photographer
265                     if (replaceExisting) {
266                         // Normal functionality - remove the existing one - we're doing an update
267                         photographer.removeImage(existing);
268                         imageDao.remove(existing);
269                     } else {
270                         // Special case - e.g. we don't want to replace admin overridden images at freeze.
271                         return new ImageItemInstance(existing);
272                     }
273                 }
274 
275                 image = new Image();
276                 image.setId(flickrImage.getFlickrId());
277                 image.setMediumImage(flickrImage.getImageUrl());
278                 image.setLargeImage(flickrImage.getLargeImageUrl());
279                 image.setPage(flickrImage.getUrl());
280                 image.setTitle(flickrImage.getTitle());
281                 image.setPostedDate(flickrImage.getPostedDate());
282 
283                 photographer.addImage(image);
284 
285                 photographyDao.persist(photographer);
286 
287                 challenge.addImage(image);
288 
289                 challengeDao.persist(challenge);
290             } else {
291                 if (photographerCheckId != null) {
292                     throw new ServiceException("photographer.mismatch");
293                 }
294             }
295         } else {
296             if (logger.isLoggable(Level.INFO)) {
297                 logger.info("Refreshing image " + flickrImage.getFlickrId());
298             }
299 
300             image.setTitle(flickrImage.getTitle());
301             image.setPostedDate(flickrImage.getPostedDate());
302             image.setPage(flickrImage.getUrl());
303             image.setMediumImage(flickrImage.getImageUrl());
304             image.setLargeImage(flickrImage.getLargeImageUrl());
305 
306             imageDao.persist(image);
307         }
308 
309         return new ImageItemInstance(image);
310     }

312     /**
313      * Method checkForExistingImage checks to see if the photographer has an image for the given challenge
314      *
315      * @param photographer of type Photographer
316      * @param tag          of type String
317      * @return Image
318      */
319     private Image checkForExistingImage(Photographer photographer, String tag) {
320         for (Image image : photographer.getImages()) {
321             if (image.getChallenge().getTag().equals(tag)) {
322                 return image;
323             }
324         }
325         return null;
326     }

366     /**
367      * Method freezeChallenge - takes the current challenge - performs a flickr search and stores the results locally.
368      */
369     public void freezeChallenge() {
370         if (logger.isLoggable(Level.INFO)) {
371             logger.info("Freezing challenge");
372         }
373 
374         Challenge challenge = challengeDao.getVotingChallenge();
375 
376         if (challenge == null) {
377             if (logger.isLoggable(Level.INFO)) {
378                 logger.info("No voting challenge found to freeze");
379             }
380 
381             return;
382         }
383 
384         if (logger.isLoggable(Level.INFO)) {
385             logger.info("Freezing challenge : " + challenge);
386         }
387 
388         FlickrImages images = flickrImageTagSearchDao.searchTag(challenge.getTag(), challenge.getStartDate());
389 
390         for (FlickrImage image : images.getImages()) {
391             retrieveAndStoreImage(image.getFlickrId(), challenge.getTag(), false, null);
392         }
393     }

471     /**
472      * Method getChallengeImages retrieves all images for a challenge for the given photographer
473      *
474      * @param tag            of type String
475      * @param photographerId of type String
476      * @return ImageItem
477      */
478     public Set<ImageItem> getChallengeImages(String tag, String photographerId) {
479         Set<ImageItem> images = new HashSet<ImageItem>();
480 
481         Challenge challenge = challengeDao.findByTag(tag);
482 
483         if (challenge == null) {
484             return null;
485         }
486 
487         if (challenge.getVotingState() != ChallengeState.OPEN) {
488             return null;
489         }
490 
491         // Grab images from flickr
492         FlickrImages flickrImages = flickrImageTagSearchDao.searchTag(tag, challenge.getStartDate());
493 
494         for (FlickrImage image : flickrImages.getImages()) {
495             if (image.getPhotographer().getFlickrId().equals(photographerId)) {
496                 ImageItem imageItem = new ImageItemInstance(image);
497                 images.add(imageItem);
498             }
499         }
500 
501         return images;
502     }
