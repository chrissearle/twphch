61      public Set<ChallengeSummary> getChallengesByType(ChallengeType type) {
62          Set<ChallengeSummary> challenges = new HashSet<ChallengeSummary>();
63  
64          switch (type) {
65              case ALL:
66                  // Get all challenges
67  
68                  for (Challenge challenge : challengeDao.getAll()) {
69                      challenges.add(new ChallengeSummaryInstance(challenge));
70                  }
71  
72                  break;
73              case OPEN:
74                  // Get open challenges
75  
76                  Challenge currentChallenge = challengeDao.getCurrentChallenge();
77                  if (currentChallenge != null) {
78                      challenges.add(new ChallengeSummaryInstance(currentChallenge));
79                  }
80  
81                  break;
82              case VOTING:
83                  // Get voting challenges
84  
85                  Challenge votingChallenge = challengeDao.getVotingChallenge();
86                  if (votingChallenge != null) {
87                      challenges.add(new ChallengeSummaryInstance(votingChallenge));
88                  }
89  
90                  break;
91              case CLOSED:
92                  // Get closed challenges
93  
94                  for (Challenge challenge : challengeDao.getClosedChallenges()) {
95                      challenges.add(new ChallengeSummaryInstance(challenge));
96                  }
97  
98                  break;
99          }
100 
101         return challenges;
102     }

104     private void doRanking(List<Image> images) {
105         long rank = 0;
106         long lastSeenValue = Long.MAX_VALUE;
107 
108         Collections.sort(images, new Comparators.ImageSortByFinalVoteCount());
109 
110         for (Image image : images) {
111             if (image.getFinalVoteCount() < lastSeenValue) {
112                 lastSeenValue = image.getFinalVoteCount();
113                 rank++;
114             }
115             image.setFinalRank(rank);
116             imageDao.persist(image);
117         }
118     }

229     public ChallengeSummary announceResults() {
230         Challenge votedChallenge = challengeDao.getVotedChallenge();
231 
232         if (votedChallenge == null) {
233             if (logger.isDebugEnabled()) {
234                 logger.debug("No challenge found to handle results");
235             }
236 
237             return null;
238         }
239 
240         ChallengeSummary challenge = new ChallengeSummaryInstance(votedChallenge);
241 
242         if (logger.isInfoEnabled()) {
243             logger.info("Results for " + challenge);
244         }
245 
246         // Sum up votes
247         List<Image> images = votedChallenge.getImages();
248 
249         for (Image image : images) {
250             image.setFinalVoteCount((long) image.getVotes().size());
251             imageDao.persist(image);
252         }
253 
254         photographyDao.clearVotes();
255 
256         doRanking(images);
257 
258         String resultsUrl = challengeMessageService.getResultsUrl(challenge);
259 
260         try {
261             twitterService.twitter(challengeMessageService.getResultsTwitter(challenge, resultsUrl));
262         } catch (TwitterServiceException tse) {
263             if (logger.isEnabledFor(Level.WARN)) {
264                 logger.warn("Unable to post to twitter" + tse.getMessage(), tse);
265             }
266         }
267 
268         List<ImageItem> imageResults = new ArrayList<ImageItem>();
269 
270         for (Image image : images) {
271             imageResults.add(new ImageItemInstance(image));
272         }
273 
274         StringBuilder messageGold = new StringBuilder();
275         StringBuilder messageSilver = new StringBuilder();
276         StringBuilder messageBronze = new StringBuilder();
277 
278         for (ImageItem imageItem : imageResults) {
279             String badgeText = "";
280 
281             String forumPost = challengeMessageService.getResultsForumSingle(imageItem);
282 
283             if (imageItem.getRank() == 1) {
284                 // Gold
285                 badgeText = challengeMessageService.getBadgeText(1, challengeMessageService.getGoldBadgeUrl(), challenge);
286                 messageGold.append(forumPost);
287             }
288             if (imageItem.getRank() == 2) {
289                 // Silver
290                 badgeText = challengeMessageService.getBadgeText(2, challengeMessageService.getSilverBadgeUrl(), challenge);
291                 messageSilver.append(forumPost);
292             }
293             if (imageItem.getRank() == 3) {
294                 // Bronze
295                 badgeText = challengeMessageService.getBadgeText(3, challengeMessageService.getBronzeBadgeUrl(), challenge);
296                 messageBronze.append(forumPost);
297             }
298             if (!"".equals(badgeText)) {
299                 try {
300                     flickrService.postComment(imageItem.getId(), badgeText);
301                 } catch (FlickrServiceException fse) {
302                     if (logger.isEnabledFor(Level.WARN)) {
303                         logger.warn("Unable to post to flickr" + fse.getMessage(), fse);
304                     }
305                 }
306             }
307         }
308 
309         String messageText = challengeMessageService.getResultsForumText(resultsUrl, messageGold.toString(), messageSilver.toString(), messageBronze.toString());
310 
311         try {
312             flickrService.postForum(challengeMessageService.getResultsForumTitle(challenge), messageText);
313         } catch (FlickrServiceException fse) {
314             if (logger.isEnabledFor(Level.WARN)) {
315                 logger.warn("Unable to post to flickr" + fse.getMessage(), fse);
316             }
317         }
318 
319         return challenge;
320     }
