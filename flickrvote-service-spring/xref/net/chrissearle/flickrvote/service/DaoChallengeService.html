105     /**
106      * Method getChallengesByType returns all challenges of the given type
107      *
108      * @param type of type ChallengeType
109      * @return Set<ChallengeSummary>
110      */
111     public Set<ChallengeSummary> getChallengesByType(ChallengeType type) {
112         Set<ChallengeSummary> challenges = new HashSet<ChallengeSummary>();
113 
114         switch (type) {
115             case ALL:
116                 // Get all challenges
117 
118                 for (Challenge challenge : challengeDao.getAll()) {
119                     challenges.add(new ChallengeSummaryInstance(challenge));
120                 }
121 
122                 break;
123             case OPEN:
124                 // Get open challenges
125 
126                 Challenge currentChallenge = challengeDao.getCurrentChallenge();
127                 if (currentChallenge != null) {
128                     challenges.add(new ChallengeSummaryInstance(currentChallenge));
129                 }
130 
131                 break;
132             case VOTING:
133                 // Get voting challenges
134 
135                 Challenge votingChallenge = challengeDao.getVotingChallenge();
136                 if (votingChallenge != null) {
137                     challenges.add(new ChallengeSummaryInstance(votingChallenge));
138                 }
139 
140                 break;
141             case CLOSED:
142                 // Get closed challenges
143 
144                 for (Challenge challenge : challengeDao.getClosedChallenges()) {
145                     challenges.add(new ChallengeSummaryInstance(challenge));
146                 }
147 
148                 break;
149         }
150 
151         return challenges;
152     }

154     /**
155      * Method doRanking calculates rank based on score
156      *
157      * @param images of type List<Image>
158      */
159     private void doRanking(List<Image> images) {
160         long rank = 0;
161         long lastSeenValue = Long.MAX_VALUE;
162 
163         Collections.sort(images, new Comparators.ImageSortByFinalVoteCount());
164 
165         for (Image image : images) {
166             if (image.getFinalVoteCount() < lastSeenValue) {
167                 lastSeenValue = image.getFinalVoteCount();
168                 rank++;
169             }
170 
171             if (logger.isLoggable(Level.INFO)) {
172                 logger.info("Setting final rank for " + image.getId() + " to " + rank);
173             }
174             image.setFinalRank(rank);
175             imageDao.persist(image);
176         }
177     }

312     /**
313      * Method announceResults announces the results of a given challenge
314      *
315      * @return ChallengeSummary
316      */
317     public ChallengeSummary announceResults() {
318         Challenge votedChallenge = challengeDao.getVotedChallenge();
319 
320         if (votedChallenge == null) {
321             if (logger.isLoggable(Level.INFO)) {
322                 logger.info("No challenge found to handle results");
323             }
324 
325             return null;
326         }
327 
328         ChallengeSummary challenge = new ChallengeSummaryInstance(votedChallenge);
329 
330         if (logger.isLoggable(Level.INFO)) {
331             logger.info("Results for " + challenge);
332         }
333 
334         // Sum up votes
335         List<Image> images = votedChallenge.getImages();
336 
337         for (Image image : images) {
338             if (logger.isLoggable(Level.INFO)) {
339                 logger.info("Setting final vote count for " + image.getId() + " to " + image.getVotes().size());
340             }
341             image.setFinalVoteCount((long) image.getVotes().size());
342             imageDao.persist(image);
343         }
344 
345         photographyDao.clearVotes();
346 
347         doRanking(images);
348 
349         String resultsUrl = challengeMessageService.getResultsUrl(challenge);
350 
351         try {
352             tweetService.tweet(challengeMessageService.getResultsTwitter(challenge, resultsUrl));
353         } catch (TwitterServiceException tse) {
354             mailService.sendPost(tse.getMessage(), tse.getTwitterMessage());
355             if (logger.isLoggable(Level.WARNING)) {
356                 logger.warning("Unable to post to twitter" + tse.getMessage());
357             }
358         }
359 
360         List<ImageItem> imageResults = new ArrayList<ImageItem>();
361 
362         for (Image image : images) {
363             imageResults.add(new ImageItemInstance(image));
364         }
365 
366         StringBuilder messageGold = new StringBuilder();
367         StringBuilder messageSilver = new StringBuilder();
368         StringBuilder messageBronze = new StringBuilder();
369 
370         for (ImageItem imageItem : imageResults) {
371             String badgeText = "";
372 
373             String forumPost = challengeMessageService.getResultsForumSingle(imageItem);
374 
375             if (imageItem.getRank() == 1) {
376                 // Gold
377                 badgeText = challengeMessageService.getBadgeText(1, challengeMessageService.getGoldBadgeUrl(), challenge);
378                 messageGold.append(forumPost);
379             }
380             if (imageItem.getRank() == 2) {
381                 // Silver
382                 badgeText = challengeMessageService.getBadgeText(2, challengeMessageService.getSilverBadgeUrl(), challenge);
383                 messageSilver.append(forumPost);
384             }
385             if (imageItem.getRank() == 3) {
386                 // Bronze
387                 badgeText = challengeMessageService.getBadgeText(3, challengeMessageService.getBronzeBadgeUrl(), challenge);
388                 messageBronze.append(forumPost);
389             }
390             if (!"".equals(badgeText)) {
391                 try {
392                     if (logger.isLoggable(Level.INFO)) {
393                         logger.info("Posting comment on " + imageItem.getId() + " with text " + badgeText);
394                     }
395                     commentDAO.postComment(imageItem.getId(), badgeText);
396                 } catch (FlickrServiceException fse) {
397                     if (logger.isLoggable(Level.WARNING)) {
398                         logger.warning("Unable to post to flickr" + fse.getMessage());
399                     }
400                 }
401             }
402         }
403 
404         String messageText = challengeMessageService.getResultsForumText(resultsUrl, messageGold.toString(), messageSilver.toString(), messageBronze.toString());
405 
406         if (logger.isLoggable(Level.INFO)) {
407             logger.info("Posting admin mail with text " + messageText);
408         }
409 
410         try {
411             mailService.sendPost(challengeMessageService.getResultsForumTitle(challenge), messageText);
412         } catch (FlickrServiceException fse) {
413             if (logger.isLoggable(Level.WARNING)) {
414                 logger.warning("Unable to post to flickr" + fse.getMessage());
415             }
416         }
417 
418         mailService.sendPost(challengeMessageService.getFrontPageTitle(), winnerService.getFrontPageHtml());
419 
420         return challenge;
421     }

512      * @param tag       of type String
513      * @param title     of type String
514      * @param startDate of type Date
515      * @param voteDate  of type Date
516      * @param endDate   of type Date

522             challenge.setName(title);

524             challenge.setVotingOpenDate(voteDate);
525             challenge.setDescription(notes);
526             challenge.setEndDate(endDate);
527         } else {
528             challenge = new Challenge(tag, title, startDate, voteDate, endDate);
529             challenge.setDescription(notes);
530         }
531 
532         challengeDao.persist(challenge);
533     }
