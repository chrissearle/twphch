17  package net.chrissearle.flickrvote.service.model.impl;
18  
19  import net.chrissearle.flickrvote.flickr.FlickrImage;
20  import net.chrissearle.flickrvote.flickr.FlickrImageStatus;
21  import net.chrissearle.flickrvote.service.model.ImageItem;
22  import net.chrissearle.flickrvote.service.model.ImageItemStatus;
23  import net.chrissearle.flickrvote.service.model.Status;
24  
25  import java.util.HashSet;
26  import java.util.Set;
27  
28  /**
29   * Class ImageItemStatusInstance implements ImageItemStatus
30   *
31   * @author chris
32   *         Created on Aug 23, 2009
33   */
34  public class ImageItemStatusInstance implements ImageItemStatus {
35      private Status status;
36      private Set<ImageItem> images;
37  
38      /**
39       * Constructor ImageItemStatusInstance creates a new ImageItemStatusInstance instance from a flickr image status.
40       *
41       * @param flickrStatus of type FlickrImageStatus
42       */
43      public ImageItemStatusInstance(FlickrImageStatus flickrStatus) {
44          switch (flickrStatus.getStatus()) {
45              case TAKEN_DATE:
46                  status = Status.TAKEN_DATE_TOO_EARLY;
47                  break;
48              case MULTIPLE_IMAGES:
49                  status = Status.MULTIPLE_IMAGE_BY_SAME_PHOTOGRAPHER;
50                  break;
51          }
52  
53          images = new HashSet<ImageItem>(flickrStatus.getImageCount());
54  
55          for (FlickrImage image : flickrStatus.getImages()) {
56              images.add(new ImageItemInstance(image));
57          }
58      }
59  
60      /**
61       * Method getStatus returns the status.
62       *
63       * @return the status.
64       */
65      public Status getStatus() {
66          return status;
67      }
68  
69      /**
70       * Method getImages returns the images affected.
71       *
72       * @return the images.
73       */
74      public Set<ImageItem> getImages() {
75          return images;
76      }
77  }
